2. Python 基础语法

2.1 数据类型

Python 是动态类型语言，变量无需声明类型；而 C 是静态类型语言，必须在使用前声明。以下是两者的数据类型对比：

C 类型 Python 类型 说明
int int Python 无溢出，支持任意精度
float / double float Python 只有 float（64 位双精度）
char / char[] str Python 字符串不可变，支持 Unicode
int (0/1) bool True / False（首字母大写）
NULL None Python 的空值
无 list 动态数组，类似 C 的数组但更强大
无 dict 键值对，类似 C 的哈希表
Python：动态类型，无需声明
x = 10
y = 3.14
name = "hello"
整数 int —— 无溢出！
a = 42
big = 2 ** 100  # C 语言早就溢出了
print(f"a = {a}, type = {type(a)}")
print(f"2^100 = {big}")
链式比较
x = 5
print(f"1 < x < 10: {1 < x < 10}")  # 等价于 1 < x and x < 10

2.2 输入输出

操作 C 语言 Python
输出 printf("x=%d", x) print(f"x={x}")
输入 scanf("%d", &x) x = input("提示")
格式化 %d, %f, %s f-string: f"{变量}"
Python 的 input() 总是返回字符串，需要手动转换类型。
输入与类型转换
name = input("请输入姓名: ")
age = int(input("请输入年龄: "))

print(f"{name} 今年 {age} 岁")
格式化输出
name = "Alice"
age = 20
score = 95.5

# f-string（推荐，Python 3.6+）
print(f"姓名: {name}, 年龄: {age}, 成绩: {score}")
格式化控制
print(f"成绩: {score:.1f}")           # 1位小数
print(f"百分比: {0.856:.1%}")         # 百分比
print(f"补零: {42:05d}")              # 00042
print(f"对齐: {'左对齐':<10}|{'右对齐':>10}|{'居中':^10}")

2.3 条件语句

C 语言
if (score >= 90) {
    printf("优秀");
} else if (score >= 60) {
    printf("及格");
} else {
    printf("不及格");
}
Python：无括号、无花括号，靠缩进
if score >= 90:
    print("优秀")
elif score >= 60:
    print("及格")
else:
    print("不及格")

Python 的 truthy / falsy 规则

Falsy（假值） Truthy（真值）
False, 0, 0.0 True, 非零数字
"" 空字符串 非空字符串
[], (), {} 空容器 非空容器
None 其他所有对象

match-case 模式匹配（Python 3.10+）

类似 C 的 switch-case，但功能更强大——可以匹配结构。
C：switch-case
switch(code) {
    case 200: printf("OK"); break;
    case 404: printf("Not Found"); break;
    default: printf("Unknown");
}
Python：match-case
match code:
    case 200:
        print("OK")
    case 404:
        print("Not Found")
    case _:
        print("Unknown")
match-case 完整示例
def handle_status(code):
    match code:
        case 200:
            return "OK"
        case 301 | 302:
            return "重定向"
        case 404:
            return "未找到"
        case 500:
            return "服务器错误"
        case _:
            return f"未知状态码: {code}"

for code in [200, 301, 404, 500, 418]:
    print(f"  {code} -> {handle_status(code)}")
结构化匹配（比 C switch 强大）
def process_command(cmd):
    match cmd.split():
        case ["quit"]:
            return "退出程序"
        case ["hello", name]:
            return f"你好, {name}!"
        case ["add", *nums]:
            return f"求和: {sum(int(n) for n in nums)}"
        case _:
            return f"未知命令: {cmd}"

for cmd in ["quit", "hello Alice", "add 1 2 3", "xyz"]:
    print(f"  '{cmd}' -> {process_command(cmd)}")

2.4 循环语句

for 循环

C 语言
for (int i = 0; i < 5; i++) {
    printf("%d ", i);
}
Python: for 变量 in 可迭代对象
for i in range(5):
    print(i, end=" ")

range() 函数

用法 含义 等价 C
range(5) 0,1,2,3,4 for(i=0;i<5;i++)
range(2, 8) 2,3,4,5,6,7 for(i=2;i<8;i++)
range(0, 10, 2) 0,2,4,6,8 for(i=0;i<10;i+=2)
range(5, 0, -1) 5,4,3,2,1 for(i=5;i>0;i--)
range 遍历
for i in range(5):
    print(i, end=" ")
print()
遍历列表
fruits = ["苹果", "香蕉", "橙子"]
for fruit in fruits:
    print(f"  水果: {fruit}")
enumerate —— 带索引遍历
for i, fruit in enumerate(fruits):
    print(f"  [{i}] {fruit}")
zip —— 并行遍历
names = ["Alice", "Bob", "Charlie"]
scores = [90, 85, 78]
for name, score in zip(names, scores):
    print(f"  {name}: {score}分")
while 循环
count = 0
while count < 5:
    print(count, end=" ")
    count += 1
print()
break / continue
print("找到第一个偶数:")
for i in range(1, 20):
    if i % 2 != 0:
        continue
    print(f"  找到: {i}")
    break
for-else（Python 特有！）
# else 在循环正常结束（没有 break）时执行
target = 7
for i in range(2, target):
    if target % i == 0:
        print(f"{target} 不是质数，因子: {i}")
        break
else:
    print(f"{target} 是质数")

3. 基本数据结构

类型 特点 C 语言类比 示例
str 不可变，有序 char[]（但不可变） "hello"
list 可变，有序，可重复 动态数组 [1, 2, 3]
tuple 不可变，有序 无 (1, 2, 3)
dict 可变，键值对，键唯一 哈希表 {"a": 1}
set 可变，无序，元素唯一 无 {1, 2, 3}

3.1 字符串 str

字符串不可变。所有「修改」操作都返回新字符串。
索引与切片
s = "Hello, Python!"

print(f"s[0] = '{s[0]}'")
print(f"s[-1] = '{s[-1]}'")
print(f"s[0:5] = '{s[0:5]}'")
print(f"s[7:] = '{s[7:]}'")
print(f"反转: '{s[::-1]}'")
upper / lower
print(f"upper: '{s.upper()}'")
print(f"lower: '{s.lower()}'")
split / join
csv_line = "Alice,90,A"
parts = csv_line.split(",")
print(f"split: {parts}")

joined = " | ".join(parts)
print(f"join: {joined}")
strip
print(f"strip: '{'  hello  '.strip()}'")
replace
print(f"replace: '{s.replace('Python', 'World')}'")
find
print(f"find: {s.find('Python')}")
startswith
print(f"startswith: {s.startswith('Hello')}")

3.2 列表 list

最常用的数据结构，类似 C 的动态数组，但可以存储混合类型。
增
nums = [1, 2, 3, 4, 5]

nums.append(6)          # 末尾添加
nums.insert(0, 0)       # 指定位置插入
nums.extend([7, 8])     # 批量添加
print(f"增: {nums}")
删
nums.pop()              # 删除末尾
nums.pop(0)             # 删除指定位置
nums.remove(3)          # 删除指定值（第一个）
print(f"删: {nums}")
改
nums[0] = 100
print(f"改: {nums}")
查
print(f"索引: nums[0] = {nums[0]}")
print(f"切片: nums[1:3] = {nums[1:3]}")
print(f"长度: {len(nums)}")
print(f"包含: {5 in nums}")
排序
nums = [3, 1, 4, 1, 5, 9, 2, 6]
print(f"sorted: {sorted(nums)}")           # 返回新列表
print(f"原列表: {nums}")                     # 不变
nums.sort()                                  # 就地排序
print(f"sort后: {nums}")
逆序
print(f"reversed: {list(reversed(nums))}")
解包
first, *middle, last = [1, 2, 3, 4, 5]
print(f"first={first}, middle={middle}, last={last}")

3.3 元组 tuple

不可变的列表。用于不应被修改的数据，如坐标、数据库记录。
基本用法与解包
point = (3, 4)
print(f"point = {point}, type = {type(point)}")

x, y = point
print(f"x={x}, y={y}")
交换变量（Python 特色！）
a, b = 1, 2
a, b = b, a
print(f"交换后: a={a}, b={b}")
元组作函数返回值
def min_max(lst):
    return min(lst), max(lst)

lo, hi = min_max([3, 1, 4, 1, 5])
print(f"最小={lo}, 最大={hi}")
命名元组（更可读）
from collections import namedtuple
Point = namedtuple("Point", ["x", "y"])
p = Point(3, 4)
print(f"p.x={p.x}, p.y={p.y}")

3.4 字典 dict

键值对映射，查找速度 O(1)。Django 中到处都是字典（请求参数、配置、上下文等）。
增 / 改
student = {"name": "Alice", "age": 20, "score": 90}

student["email"] = "alice@example.com"
student["age"] = 21
print(f"字典: {student}")
查
print(f"name: {student['name']}")
print(f"get: {student.get('phone', '未设置')}")  # 安全获取
删
del student["email"]
print(f"删除后: {student}")
遍历
for key, value in student.items():
    print(f"  {key}: {value}")
常用方法
print(f"keys: {list(student.keys())}")
print(f"values: {list(student.values())}")
字典合并（Python 3.9+）
defaults = {"theme": "light", "lang": "en", "page_size": 20}
user_prefs = {"theme": "dark", "lang": "zh"}
merged = defaults | user_prefs     # 后者覆盖前者
print(f"合并: {merged}")
setdefault
d = {"a": 1}
d.setdefault("b", 2)   # 不存在则设置
d.setdefault("a", 99)  # 已存在则不变
print(f"setdefault: {d}")
defaultdict
from collections import defaultdict
groups = defaultdict(list)
for name, dept in [("Alice", "技术"), ("Bob", "市场"), ("Charlie", "技术")]:
    groups[dept].append(name)
print(f"分组: {dict(groups)}")

3.5 集合 set

元素唯一，无序。用于去重和集合运算。
去重
nums = [1, 2, 2, 3, 3, 3, 4]
unique = set(nums)
print(f"去重: {unique}")
集合运算
a = {1, 2, 3, 4, 5}
b = {4, 5, 6, 7, 8}
print(f"交集: {a & b}")
print(f"并集: {a | b}")
print(f"差集: {a - b}")
print(f"对称差集: {a ^ b}")
成员判断（比 list 快很多）
print(f"3 in a: {3 in a}")
