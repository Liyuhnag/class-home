Python 文件操作 — 交互演示

始终用 with open(...)，它保证文件自动关闭，即使发生异常。

1. 文件打开模式

模式说明文件不存在时
'r'只读报错
'w'只写（清空）创建
'a'追加创建
'x'创建写（已存在报错）创建
'rb'/'wb'二进制读/写同上
📋文件读写代码
# 写入（清空重写）
with open('demo.txt', 'w', encoding='utf-8') as f:
    f.write('第一行内容\n')
    f.write('第二行内容\n')

# 追加
with open('demo.txt', 'a', encoding='utf-8') as f:
    f.write('追加一行\n')

# 读取全部
with open('demo.txt', 'r', encoding='utf-8') as f:
    content = f.read()

# 逐行读取
with open('demo.txt', 'r', encoding='utf-8') as f:
    for line in f:          # 内存友好，适合大文件
        print(line.strip())

2. JSON 文件处理

📋JSON 读写代码
import json

# 写入 JSON
data = {'name': 'Alice', 'age': 20, 'courses': ['Python', 'Math'], 'graduated': False}
with open('student.json', 'w', encoding='utf-8') as f:
    json.dump(data, f, ensure_ascii=False, indent=2)

# 读取 JSON
with open('student.json', 'r', encoding='utf-8') as f:
    loaded = json.load(f)
print(loaded['name'])      # Alice
print(loaded['courses'])   # ['Python', 'Math']

# 修改后写回
loaded['age'] += 1
loaded['courses'].append('Django')
with open('student.json', 'w', encoding='utf-8') as f:
    json.dump(loaded, f, ensure_ascii=False, indent=2)

3. CSV 处理

📋CSV 读写代码
import csv

# 写入 CSV
students = [
    {'name': 'Alice', 'age': 20, 'score': 90},
    {'name': 'Bob',   'age': 21, 'score': 85},
]
with open('students.csv', 'w', newline='', encoding='utf-8') as f:
    writer = csv.DictWriter(f, fieldnames=['name', 'age', 'score'])
    writer.writeheader()
    writer.writerows(students)

# 读取 CSV
with open('students.csv', 'r', encoding='utf-8') as f:
    reader = csv.DictReader(f)
    for row in reader:
        print(row['name'], row['score'])  # Alice 90

4. 上下文管理器

💡 C 的 fopen/fclose vs Python with C 需要手动 fclose，Python with 自动调用 __exit__，即使发生异常也会关闭资源。
本质：with 进入时调用 __enter__，离开时（无论是否异常）调用 __exit__。
📋with 语句工作原理
# with 语句等价展开
with open('demo.txt', 'r') as f:
    data = f.read()

# 等价于：
f = open('demo.txt', 'r')
try:
    f.__enter__()        # 返回文件对象本身
    data = f.read()
finally:
    f.__exit__(None, None, None)  # 关闭文件

# ✅ 即使 f.read() 抛出异常，文件也会被关闭
📋两种实现方式对比

方式一：类（完整控制）

class ManagedFile:
    def __init__(self, path, mode):
        self.path = path
        self.mode = mode

    def __enter__(self):
        print('打开文件')
        self.f = open(self.path, self.mode)
        return self.f       # as 变量

    def __exit__(self, exc_type,
                 exc_val, exc_tb):
        self.f.close()
        print('文件已关闭')
        return False  # 不压制异常

with ManagedFile('a.txt','w') as f:
    f.write('hello')

方式二：@contextmanager（简洁）

from contextlib import contextmanager
import time

@contextmanager
def timer(name=''):
    t0 = time.perf_counter()
    print(f'[{name}] 开始')
    try:
        yield        # with 块在此执行
    finally:
        elapsed = time.perf_counter() - t0
        print(f'[{name}] 耗时: {elapsed:.4f}s')

with timer('计算'):
    result = sum(range(1_000_000))
    # [计算] 开始
    # [计算] 耗时: 0.031s
🎯 __exit__ 的三个参数： exc_type（异常类型）、exc_val（异常值）、exc_tb（traceback）。 无异常时三个都是 None。返回 True 表示压制异常（不再向上传播），返回 False 正常传播。
